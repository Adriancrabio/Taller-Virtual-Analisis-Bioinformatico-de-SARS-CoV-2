Sub color_aaSubs()
'
' color_aaSubs Macro
'




    Cells.Select
    Selection.FormatConditions.Add Type:=xlTextString, String:="S:E484K", _
        TextOperator:=xlContains

        .Color = -16383844
        .TintAndShade = 0
    End With

        .TintAndShade = 0
